-- Creating trigger  for  'Ac_Ata'
CREATE Trigger "UC_AMD"."TRG_AC_ATA" 
before insert on "UC_AMD"."AC_ATA"
for each row
begin
select "UC_AMD"."SEQ_AC_ATA".nextval into:new.id from dual;
end;
